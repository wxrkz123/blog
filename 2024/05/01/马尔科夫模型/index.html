Data analysis.py

import pandas as pd
import statsmodels.api as sm
from statsmodels.tsa.stattools import adfuller
from docx import Document
import mplfinance as mpf

# 读取数据
data = pd.read_excel('E:\\MZL.DCE豆粕主连.xlsx')
data['Date'] = pd.to_datetime(data['Date'])  # 确保日期格式正确
# 确保日期列是日期类型，并设置为索引
data.set_index('Date', inplace=True)

# 确保数据按日期排序
data.sort_index(inplace=True)

# 描述性统计
desc_stats = data['Close'].describe()

# 白噪声检验 - Ljung-Box
Ljung_Box_Result = sm.stats.acorr_ljungbox(data['Close'], lags=[10], return_df=False)

# 增广DF单位根检验
adf_result = adfuller(data['Close'], autolag='AIC')

# 创建Word文档
doc = Document()
doc.add_heading('Financial Data Analysis Report', level=1)
doc.add_heading('Descriptive Statistics', level=2)
doc.add_paragraph(f"Count: {desc_stats['count']}\nMean: {desc_stats['mean']}\nStd Dev: {desc_stats['std']}"
                  f"\nMin: {desc_stats['min']}\n25%: {desc_stats['25%']}\nMedian: {desc_stats['50%']}"
                  f"\n75%: {desc_stats['75%']}\nMax: {desc_stats['max']}")

doc.add_heading('White Noise Test (Ljung-Box)', level=2)
doc.add_paragraph(f"Lag 10 Statistic: {Ljung_Box_Result['lb_pvalue'].values[0]}, P-value: {Ljung_Box_Result['lb_stat'].values[0]}")

doc.add_heading('Augmented Dickey-Fuller Test', level=2)
doc.add_paragraph(f"ADF Statistic: {adf_result[0]}\nP-value: {adf_result[1]}"
                  f"\nCritical Values: {adf_result[4]}")

# 创建一个K线图
mpf_style = mpf.make_mpf_style(base_mpf_style='charles', rc={'font.size': 12})
mpf_plot = {
    'type': 'candle',
    'mav': (3, 6, 9),  # 3, 6, 9日移动平均线
    'volume': True,
    'ylabel': 'Price',
    'ylabel_lower': 'Volume',
    'figratio': (18, 10),  # 图表大小比例
    'figscale': 1.5         # 图表放大系数
}

# 按年分组并为每一年生成K线图
for year, group in data.groupby(data.index.year):
    title = f'Detailed Daily Candlestick Chart for {year}'
    mpf_plot['title'] = title
    fig_path = f'detailed_kline_chart_{year}.png'  # 图片保存路径
    mpf.plot(group, **mpf_plot, style=mpf_style, savefig=fig_path)
    
    # 将图片添加到Word文档
    doc.add_page_break()
    doc.add_heading(f'Daily Close Price Chart {year}', level=2)
    doc.add_picture(fig_path)

# 保存Word文档
doc.save('financial_data_report.docx')

# 整理所有结果为DataFrame，以便导出到Excel
results_df = pd.DataFrame({
    "Statistic": ["Count", "Mean", "Std Dev", "Min", "25%", "Median", "75%", "Max", 
                  "Ljung-Box Statistic", "Ljung-Box P-value", "ADF Statistic", "ADF P-value", "ADF Critical Value (1%)", "ADF Critical Value (5%)", "ADF Critical Value (10%)"],
    "Value": [desc_stats['count'], desc_stats['mean'], desc_stats['std'], desc_stats['min'],
              desc_stats['25%'], desc_stats['50%'], desc_stats['75%'], desc_stats['max'],
              Ljung_Box_Result['lb_pvalue'].values[0], Ljung_Box_Result['lb_stat'].values[0], adf_result[0], adf_result[1], adf_result[4]['1%'], adf_result[4]['5%'], adf_result[4]['10%']]
})

# 将原始数据和计算结果合并到同一个Excel文件
with pd.ExcelWriter('processed_financial_data.xlsx') as writer:
    data.to_excel(writer, sheet_name='Original Data', index=False)
    results_df.to_excel(writer, sheet_name='Analysis Results', index=False)

HMM Model Analysis.py

import pandas as pd
from hmmlearn import hmm
import matplotlib.pyplot as plt
import numpy as np
from docx import Document
from docx.shared import Inches

def get_candle_type(row):
    if row['Close'] == row['Open']:
        if row['High'] > row['Close'] and row['Low'] < row['Open']:
            return 'Doji with shadows'
        elif row['High'] > row['Close']:
            return 'Doji with upper shadow'
        elif row['Low'] < row['Open']:
            return 'Doji with lower shadow'
        return 'Doji'

    if row['Close'] > row['Open']:
        if row['High'] > row['Close'] and row['Low'] < row['Open']:
            return 'Bullish with upper and lower shadows'
        elif row['High'] > row['Close']:
            return 'Bullish with upper shadow'
        elif row['Low'] < row['Open']:
            return 'Bullish with lower shadow'
        return 'Bullish candlestick'  # No shadows

    if row['Close'] < row['Open']:
        if row['High'] > row['Open'] and row['Low'] < row['Close']:
            return 'Bearish with upper and lower shadows'
        elif row['High'] > row['Open']:
            return 'Bearish with upper shadow'
        elif row['Low'] < row['Close']:
            return 'Bearish with lower shadow'
        return 'Bearish candlestick'  # No shadows

# 数据加载
data = pd.read_excel('E:\\MZL.DCE豆粕主连.xlsx')
data['Date'] = pd.to_datetime(data['Date'])
data.sort_values('Date', inplace=True)

data['CandleType'] = data.apply(get_candle_type, axis=1)

# 状态编码
candle_mapping = {
    'Doji with shadows': 0,
    'Doji with upper shadow': 1,
    'Doji with lower shadow': 2,
    'Doji': 3,
    'Bullish with upper and lower shadows': 4,
    'Bullish with upper shadow': 5,
    'Bullish with lower shadow': 6,
    'Bullish candlestick': 7,
    'Bearish with upper and lower shadows': 8,
    'Bearish with upper shadow': 9,
    'Bearish with lower shadow': 10,
    'Bearish candlestick': 11
}

data['State'] = data['CandleType'].map(candle_mapping)

# 添加年份和月份列
data['Year'] = data['Date'].dt.year
data['Month'] = data['Date'].dt.month

# 初始化HMM模型
model = hmm.MultinomialHMM(n_components=len(candle_mapping), n_iter=100)

state_counts = data.pivot_table(index=['Year', 'Month'], columns='State', aggfunc='size', fill_value=0)

# 生成一个包含状态计数的numpy数组
states_array = state_counts.values

# 拟合HMM模型
model.fit(states_array)

# 获取状态转移矩阵
transmat = model.transmat_

# 预测下一个状态
last_month_states = states_array[-1].reshape(1, -1)
next_state = model.predict(last_month_states)
predicted_state_name = list(candle_mapping.keys())[next_state[0]]

# 创建一个Word文档
doc = Document()
doc.add_heading('HMM Model Analysis', level=1)
doc.add_paragraph(f'Predicted state for next month: {predicted_state_name}')

# 添加状态转移矩阵热图
plt.figure(figsize=(12, 10))
plt.imshow(transmat, cmap='viridis', interpolation='nearest', aspect='auto')
plt.colorbar()
plt.title('State Transition Matrix Heatmap')
plt.xlabel('Target status')
plt.ylabel('Starting state')
plt.xticks(np.arange(len(candle_mapping)), list(candle_mapping.keys()), rotation=90, fontsize=8)
plt.yticks(np.arange(len(candle_mapping)), list(candle_mapping.keys()), fontsize=8)
plt.grid(False)
plt.tight_layout()
plt.savefig('transition_matrix_heatmap.png')
doc.add_picture('transition_matrix_heatmap.png', width=Inches(6))

# 添加状态计数趋势图
plt.figure(figsize=(20, 15))
state_counts.plot(kind='line', marker='o')
plt.title('Trends of each state over time', fontsize=18)
plt.xlabel('Time (year-month)')
plt.ylabel('Status count')
plt.legend(title='State', labels=list(candle_mapping.keys()), title_fontsize=16, fontsize=14, loc='upper center', bbox_to_anchor=(0.5, -0.1), ncol=3)
plt.grid(True)
plt.tight_layout()
plt.savefig('state_counts_trends.png', dpi=300, bbox_inches='tight')
doc.add_picture('state_counts_trends.png', width=Inches(8))

# 保存文档
doc.save('HMM_Model_Analysis_Report.docx')

# 输出数据到Excel
with pd.ExcelWriter('HMM_Model_Analysis_Output.xlsx') as writer:
    data.to_excel(writer, sheet_name='Data with States')
    state_counts.to_excel(writer, sheet_name='State Counts')
    pd.DataFrame(transmat, index=list(candle_mapping.keys()), columns=list(candle_mapping.keys())).to_excel(writer, sheet_name='State Transition Matrix')
