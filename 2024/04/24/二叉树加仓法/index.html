二叉树加仓法策略分析

Binary tree addition method.py

# 导入必要的库
import pandas as pd  # 用于数据处理和分析
import matplotlib.pyplot as plt  # 用于绘制图形
import numpy as np  # 用于数值计算
from docx import Document  # 用于创建和修改Word文档

# 读取Excel文件中的数据，您需要根据实际的文件路径来修改此处
data = pd.read_excel('E:\\MZL.DCE豆粕主连.xlsx')

# 设置策略的初始参数(需要根据实际进行调参)
initial_investment = 10000  # 初始投资额度
max_investment_ratio = 0.9  # 允许的最大投资比例
add_percentage = 0.001  # 加仓时增加的投资比例
drop_threshold = 0.01  # 触发加仓的价格跌幅阈值
transaction_fee_ratio = 0.0001  # 交易手续费比例

# 初始化策略所需的变量
cash = initial_investment*100  # 现金余额
position = 0  # 持仓量
average_cost = 0  # 平均持仓成本
total_investment = initial_investment  # 总投资额度
net_asset_value = []  # 存储每个时点的净资产价值

# 用于可视化的变量列表
investment_list, position_list, cash_list, avg_cost_list, volume_list, open_price_list = [], [], [], [], [], []

# 策略的主执行循环，遍历数据集中的每一行
for index, row in data.iterrows():
    open_price = row['Open']  # 当天的开盘价
    volume = row['Volume']  # 当天的交易量

    # 如果是第一天，则使用初始投资额来买入
    if index == 0:
        position = (initial_investment * (1 - transaction_fee_ratio)) / open_price
        average_cost = open_price / (1 - transaction_fee_ratio)
        cash -= initial_investment
    else:
        # 非第一天，计算当前价格相对于平均成本的跌幅
        price_drop = (average_cost - open_price) / average_cost
        
        # 如果跌幅达到阈值，并且总投资额没有超过最大投资比例，则加仓
        if price_drop >= drop_threshold and total_investment <= cash * max_investment_ratio:
            # 计算加仓金额，取加仓比例和剩余现金的较小者
            add_cash = min(cash * add_percentage, cash * max_investment_ratio - total_investment)
            # 扣除交易手续费后的实际加仓金额
            add_cash_after_fee = add_cash * (1 - transaction_fee_ratio)
            # 加仓后新的持仓量
            new_position = add_cash_after_fee / open_price
            # 更新总持仓量和平均持仓成本
            position += new_position
            total_cost = average_cost * (position - new_position) + open_price * new_position
            average_cost = total_cost / position
            # 更新现金余额和总投资额
            cash -= add_cash
            total_investment += add_cash

    # 更新净资产价值
    current_nav = position * open_price + cash
    net_asset_value.append(current_nav)

    # 将数据记录到列表中，用于后续的可视化和分析
    investment_list.append(total_investment)
    position_list.append(position)
    cash_list.append(cash)
    avg_cost_list.append(average_cost)
    volume_list.append(volume)
    open_price_list.append(open_price)

# 创建一个新的DataFrame，用于存储策略的执行数据
strategy_data = pd.DataFrame({
    'Investment': investment_list,
    'Position': position_list,
    'Cash': cash_list,
    'Average Cost': avg_cost_list,
    'Volume': volume_list,
    'Open Price': open_price_list,
    'Net Asset Value': net_asset_value
})

# 计算总收益率、年化收益率和Sharpe比率，用于评估策略的表现
total_return = (net_asset_value[-1] - initial_investment) / initial_investment
annual_return = (total_return + 1) ** (252 / len(data)) - 1  # 假设一年252个交易日
volatility = np.std(np.diff(np.log(net_asset_value))) * np.sqrt(252)  # 波动率的年化值
sharpe_ratio = annual_return / volatility  # Sharpe比率

# 打印策略执行结果
print(f"策略执行完毕，最终持仓量：{position}, 平均成本：{average_cost}, 剩余现金：{cash}, 总投资额：{total_investment}, 总收益率：{total_return:.2%}, 年化收益率：{annual_return:.2%}, Sharpe比率：{sharpe_ratio:.2f}")

# 绘制和保存图表，展示策略的各项指标随时间的变化
plt.figure(figsize=(14, 10))
titles = ["The total amount of investment varies over time", "The volume of open interest changes over time", "Remaining cash changes over time", "The average cost changes over time", "Volume changes over time", "The opening price changes over time"]
data_lists = [investment_list, position_list, cash_list, avg_cost_list, volume_list, open_price_list]
for i, data_list in enumerate(data_lists):
    plt.subplot(3, 2, i + 1)
    plt.plot(data_list, label=titles[i])
    plt.title(titles[i])
    plt.legend()

plt.tight_layout()
plt.savefig('Analysis.png')
plt.show()

# 使用python-docx库创建一个Word文档，并添加策略的执行结果
doc = Document()
doc.add_heading('Strategy Performance', level=1)
doc.add_paragraph(f"Final Position: {position}") #最终持仓量
doc.add_paragraph(f" Average Cost: {average_cost}") #平均成本
doc.add_paragraph(f" Remaining Cash: {cash}") #剩余现金
doc.add_paragraph(f" Total Investment: {total_investment}") #总投资额
doc.add_paragraph(f" Total Return: {total_return:.2%}") #总收益率
doc.add_paragraph(f" Annualized Return: {annual_return:.2%}") #年化收益率
doc.add_paragraph(f"Sharpe Ratio: {sharpe_ratio:.2f}") #Sharpe比率

# 保存Word文档
doc.save("Strategy_performance_metrics.docx")

# 将策略过程中的数据保存到Excel文件中
strategy_data.to_excel('strategy_output_data.xlsx', index=False)
