import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller, coint
from docx import Document
import seaborn as sns
import numpy as np

def adf_test(series, name, doc):
    # 进行ADF检验
    result = adfuller(series, autolag='AIC')
    
    # 向Word文档中添加ADF统计量和p-value
    doc.add_paragraph(f"ADF Statistic for {name}: {result[0]:.4f}")
    doc.add_paragraph(f"p-value for {name}: {result[1]:.4f}")
    
    # 添加临界值表格
    table = doc.add_table(rows=1, cols=5)
    table.style = 'Table Grid'
    hdr_cells = table.rows[0].cells
    hdr_cells[0].text = 'Type'
    hdr_cells[1].text = 'Value'
    hdr_cells[2].text = '1%'
    hdr_cells[3].text = '5%'
    hdr_cells[4].text = '10%'
    row_cells = table.add_row().cells
    row_cells[0].text = f'{name}'
    row_cells[1].text = f"{result[0]:.4f}"
    row_cells[2].text = f"{result[4]['1%']}"
    row_cells[3].text = f"{result[4]['5%']}"
    row_cells[4].text = f"{result[4]['10%']}"
    
    # 设置图形
    fig, ax = plt.subplots()
    ax.plot(series, label=f'{name} Series', color='blue', linewidth=2)
    ax.set_title(f'ADF Test Result for {name}')
    ax.set_xlabel('Date')
    ax.set_ylabel('Value')
    ax.legend(loc='upper left')
    
    # 添加文本框显示ADF结果，位置调整到图形上部中央
    textstr = f'ADF Statistic: {result[0]:.4f}\nP-value: {result[1]:.4f}'
    props = dict(boxstyle='round', facecolor='wheat', alpha=0.5)
    ax.text(0.5, 0.9, textstr, transform=ax.transAxes, fontsize=12,
            verticalalignment='top', horizontalalignment='center', bbox=props)
    
    # 保存并显示图形
    fig.savefig(f'adf_test_{name}.png')
    plt.show()
    
    return result

# 添加计算最大回撤的函数
def max_drawdown(return_list):
    max_value = float('-inf')
    max_drawdown = 0
    for i in return_list:
        max_value = max(max_value, i)
        max_drawdown = min(max_drawdown, (i - max_value) / max_value if max_value != 0 else 0)
    return max_drawdown


# 添加计算夏普比率的函数
def sharpe_ratio(return_list, risk_free_rate=0.00):  
    std = np.std(return_list)
    if std == 0:
        return None
    mean = np.mean(return_list)
    sharpe_ratio = (mean - risk_free_rate) / std
    return sharpe_ratio

data = pd.read_excel('E:\\豆粕合约.xlsx')
contracts = ['m2408', 'm2409', 'm2411', 'm2412', 'm2501', 'm2503']
prices = data[['Date'] + contracts]
prices.set_index('Date', inplace=True)

doc = Document()
doc.add_heading('Inter-delivery Strategy Analysis with Six Contracts', level=1)

# 计算相关系数矩阵
correlation_matrix = prices.corr()

# 使用Seaborn库绘制热力图
plt.figure(figsize=(10, 8))
sns.heatmap(correlation_matrix, annot=True, cmap='coolwarm', fmt=".2f", linewidths=.5)
plt.title('Correlation Matrix Heatmap of Futures Contracts')
plt.savefig('Correlation Matrix Heatmap.png')
plt.show()

# 设置画布大小
plt.figure(figsize=(12, 6))

# 绘制每个合约的价格序列
for contract in contracts:
    plt.plot(prices[contract], label=contract)

# 设置图表标题和图例
plt.title('Price Series of Futures Contracts')
plt.legend()

# 保存图形
plt.savefig('price_series.png')

# 显示图形
plt.show()

for contract in contracts:
    adf_test(prices[contract], contract, doc)

pairs = []
for i in range(len(contracts)):
    for j in range(i + 1, len(contracts)):
        coint_stat, p_value, _ = coint(prices[contracts[i]], prices[contracts[j]])
        if p_value < 0.05:
            pairs.append((contracts[i], contracts[j]))
            doc.add_paragraph(f"Cointegration found between {contracts[i]} and {contracts[j]} with p-value: {p_value}")

for pair in pairs:
    spread = prices[pair[0]] - prices[pair[1]]
    mean_spread = spread.mean()
    std_spread = spread.std()
    prices['z_score'] = (spread - mean_spread) / std_spread

    plt.figure(figsize=(12, 6))
    plt.plot(prices.index, prices['z_score'], label='Z-Score of Spread')
    plt.axhline(y=2, color='r', linestyle='--', label='Upper Threshold (2σ)')
    plt.axhline(y=-2, color='g', linestyle='--', label='Lower Threshold (-2σ)')
    plt.title(f'Spread and Thresholds for {pair[0]} and {pair[1]}')
    plt.legend()
    plt.savefig(f'spread_thresholds_{pair[0]}_{pair[1]}.png')
    plt.show()
    
    # 生成交易信号
    prices['position'] = 0
    prices.loc[prices['z_score'] > 2, 'position'] = -1  
    prices.loc[prices['z_score'] < -2, 'position'] = 1  
    prices.loc[abs(prices['z_score']) < 0.5, 'position'] = 0  

    prices['position_flat'] = np.where(abs(prices['position']) == 0, 1, 0)
    prices['trade_flag'] = prices['position_flat'].diff()
    trade_count = len(prices.loc[prices['trade_flag'] == 1])  # 交易次数

    prices['daily_return'] = prices['position'].shift(1) * (prices[pair[0]].pct_change() - prices[pair[1]].pct_change())
    prices['cum_profit'] = (1 + prices['daily_return']).cumprod()
    prices['trade_return'] = np.where(prices['trade_flag'] == 1, prices['cum_profit'].shift() - 1, np.nan)
    profit_count = sum(prices['trade_return'] > 0)  # 盈利次数

    prices['cumulative_return'] = (1 + prices['daily_return']).cumprod() - 1
    plt.figure(figsize=(12, 6))
    plt.plot(prices.index, prices['cumulative_return'], label='Cumulative Returns')
    plt.title(f'Cumulative Returns from Arbitrage Strategy for {pair[0]} and {pair[1]}')
    plt.xlabel('Date')
    plt.ylabel('Cumulative Return')
    plt.legend()
    plt.savefig(f'arbitrage_cumulative_returns_{pair[0]}_{pair[1]}.png')
    plt.show()
    
    win_rate = profit_count / trade_count if trade_count > 0 else None  # 胜率 
    mdd = max_drawdown(prices['cumulative_return'])  # 最大回撤
    sharpe = sharpe_ratio(prices['daily_return'])  # 夏普比率

    # 插入Word中
    doc.add_paragraph(f'Performance Attribution metrics for pair {pair}')
    doc.add_paragraph(f'Total trades: {trade_count}')
    doc.add_paragraph(f'Number of profitable trades: {profit_count}')
    doc.add_paragraph(f'Winning rate: {win_rate if win_rate is not None else "N/A"}')
    doc.add_paragraph(f'Maximum drawdown: {mdd:.4f}')
    doc.add_paragraph(f'Sharpe ratio: {sharpe if sharpe is not None else "N/A"}')

doc.save("Inter-delivery_strategy.docx")

data.to_excel('Inter-delivery_strategy_output_data.xlsx', index=False)
