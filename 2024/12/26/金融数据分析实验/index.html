金融数据分析实验

import yfinance as yf
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import minimize
from docx import Document
from itertools import combinations
from docx.shared import Inches

# 设置显示选项
pd.options.display.float_format = '{:.4f}'.format

# ========== 1. 获取股票数据 ==========
# 选择 5 支股票
stocks = ['AAPL', 'GOOGL', 'MSFT', 'JPM', 'XOM']  
start_date = '2021-01-01'
end_date = '2023-12-31'

# 下载完整数据
data_full = yf.download(stocks, start=start_date, end=end_date)

# 提取收盘价数据
data = data_full['Close']

# 检查并填充缺失值
data.fillna(method='ffill', inplace=True)

# 导出完整数据为 CSV 文件
data_full.to_csv('stock_data_full.csv')

# ========== 2. 数据清洗与描述性统计 ==========
# 计算每日收益率
returns = data.pct_change().dropna()

# 描述性统计
doc = Document()
doc.add_heading('Descriptive statistics on stock closing prices', level=1)

for stock in stocks:
    doc.add_heading(f'{stock} Descriptive statistics', level=2)
    stats = data[stock].describe()
    doc.add_paragraph(f"mean: {stats['mean']:.4f}")
    doc.add_paragraph(f"standard deviation: {stats['std']:.4f}")
    doc.add_paragraph(f"minimum: {stats['min']:.4f}")
    doc.add_paragraph(f"25% quantile: {stats['25%']:.4f}")
    doc.add_paragraph(f"median: {stats['50%']:.4f}")
    doc.add_paragraph(f"75% quantile: {stats['75%']:.4f}")
    doc.add_paragraph(f"maximum: {stats['max']:.4f}")
    doc.add_paragraph("\n")

# 保存 Word 文件
doc.save('stock_close_price_statistics.docx')

# ========== 3. 绘制可视化图表 ==========
# 1) 收盘价时间序列图
plt.figure(figsize=(12, 6))
for stock in stocks:
    plt.plot(data[stock], label=stock)
plt.title("A time series of the stock's closing price")
plt.xlabel("Date")
plt.ylabel("Close")
plt.legend()
plt.grid()
plt.savefig("A time series of the stock's closing price.png")
plt.show()

# 2) 收益率分布直方图
for stock in stocks:
    plt.figure(figsize=(12, 6))
    plt.hist(returns[stock], bins=50, alpha=0.6, label=stock, color='blue')
    plt.title(f"{stock} Histogram of yield distribution")
    plt.xlabel("Yield")
    plt.ylabel("frequency")
    plt.legend()
    plt.grid()
    plt.savefig(f"{stock}_Histogram of yield distribution.png")  # 保存每只股票的直方图
    plt.show()

# 3) 选取两只股票绘制收益率散点图
# 获取股票的所有两两组合
stock_pairs = list(combinations(stocks, 2))  # 生成所有可能的两两组合

# 创建一个循环绘制每一对股票的散点图并保存为文件
for stock1, stock2 in stock_pairs:
    plt.figure(figsize=(8, 6))
    plt.scatter(returns[stock1], returns[stock2], alpha=0.6)
    plt.title(f"{stock1} vs {stock2} Yield correlation")
    plt.xlabel(f"{stock1} Daily yield")
    plt.ylabel(f"{stock2} Daily yield")
    plt.grid()
    plt.savefig(f"{stock1}_vs_{stock2}_scatter.png")
    plt.close()

# ========== 4. 投资组合优化与有效前沿 ==========
# 无风险收益率
rf = 0.03

# 函数：计算投资组合的收益率和波动率
def portfolio_stats(weights, mean_returns, cov_matrix):
    weights = np.array(weights)  # 将 weights 转为 NumPy 数组
    portfolio_return = np.sum(weights * mean_returns) * 252  # 年化收益率
    portfolio_std = np.sqrt(np.dot(weights.T, np.dot(cov_matrix, weights))) * np.sqrt(252)  # 年化波动率
    return portfolio_return, portfolio_std

# 函数：负的夏普比率（优化目标）
def negative_sharpe_ratio(weights, mean_returns, cov_matrix, rf):
    portfolio_return, portfolio_std = portfolio_stats(weights, mean_returns, cov_matrix)
    sharpe_ratio = (portfolio_return - rf) / portfolio_std
    return -sharpe_ratio

# 函数：约束条件
def portfolio_constraints(weights):
    return np.sum(weights) - 1  # 权重之和为1

# 优化投资组合并绘制有效前沿
sharpe_ratios = {}  # 保存每对组合的夏普比率
document = Document()  # 创建Word文档

# 遍历所有两两组合
for stock1, stock2 in stock_pairs:
    # 数据选择
    selected_returns = returns[[stock1, stock2]]  # 选择两只股票的收益率
    mean_returns = selected_returns.mean()  # 平均收益率
    cov_matrix = selected_returns.cov()  # 协方差矩阵
    
    # 初始权重
    num_assets = len(selected_returns.columns)
    init_guess = [1.0 / num_assets] * num_assets  # 初始均等分配
    bounds = tuple((0, 1) for _ in range(num_assets))  # 权重在0到1之间
    constraints = ({'type': 'eq', 'fun': portfolio_constraints})  # 权重之和为1
    
    # 优化最大夏普比率
    result = minimize(negative_sharpe_ratio, init_guess, args=(mean_returns, cov_matrix, rf),
                      method='SLSQP', bounds=bounds, constraints=constraints)
    
    # 计算最优投资组合
    optimal_weights = result.x
    max_sharpe_return, max_sharpe_std = portfolio_stats(optimal_weights, mean_returns, cov_matrix)
    max_sharpe_ratio = (max_sharpe_return - rf) / max_sharpe_std
    
    # 保存夏普比率
    sharpe_ratios[f"{stock1} & {stock2}"] = max_sharpe_ratio
    
    # 有效前沿：计算不同权重下的组合收益率和波动率
    frontier_returns = []
    frontier_stds = []
    for weight in np.linspace(0, 1, 100):
        weights = [weight, 1 - weight]  # 两只股票的权重
        portfolio_return, portfolio_std = portfolio_stats(weights, mean_returns, cov_matrix)
        frontier_returns.append(portfolio_return)
        frontier_stds.append(portfolio_std)
    
    # 绘制有效前沿
    plt.figure(figsize=(10, 6))
    plt.plot(frontier_stds, frontier_returns, label='Efficient Frontier')  # 有效前沿
    plt.scatter(max_sharpe_std, max_sharpe_return, color='red', label='Max Sharpe Ratio', marker='*', s=200)  # 最优组合
    plt.title(f"Efficient Frontier: {stock1} & {stock2}")
    plt.xlabel("Portfolio Volatility (Std Dev)")
    plt.ylabel("Portfolio Return")
    plt.legend()
    plt.grid()
    plt.savefig(f"Efficient_Frontier_{stock1}_{stock2}.png")
    plt.close()

    # 添加到Word文档
    document.add_heading(f"{stock1} & {stock2} Portfolio", level=2)
    document.add_paragraph(f"Maximum Sharpe Ratio: {max_sharpe_ratio:.4f}")
    document.add_paragraph(f"Optimal Weights: {stock1}: {optimal_weights[0]:.4f}, {stock2}: {optimal_weights[1]:.4f}")
    document.add_picture(f"Efficient_Frontier_{stock1}_{stock2}.png", width=Inches(5))
    document.add_paragraph("\n")

# ========== 5. 输出夏普比率到Word文件 ==========
# 添加一个总结表格
document.add_heading("Summary of Sharpe Ratios", level=1)
table = document.add_table(rows=1, cols=2)  # 创建表格
table.style = 'Table Grid'

# 表头
hdr_cells = table.rows[0].cells
hdr_cells[0].text = 'Stock Pair'
hdr_cells[1].text = 'Sharpe Ratio'

# 填充表格数据
for pair, sharpe in sharpe_ratios.items():
    row_cells = table.add_row().cells
    row_cells[0].text = pair
    row_cells[1].text = f"{sharpe:.4f}"

# 保存Word文档
document.save("Portfolio_Analysis.docx")
