# 导入必要的库
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import pandas_ta as ta
from matplotlib.pyplot import MultipleLocator

# 读取excel文件中的数据，E盘路径下的一个数据文件
data = pd.read_excel('E:\\课程资料\\大三（下）\\量化投资管理\\分析作业\\MZL.DCE豆粕主连.xlsx')

# 时间字段格式化为pandas datetime格式，以便于后续时间序列分析
data['Date'] = pd.to_datetime(data['Date'])

# 计算移动平均收敛/发散指标 (MACD: Moving Average Convergence Divergence)
data['MACD'], data['MACD_Signal'], data['MACD_Hist'] = ta.macd(data['Close'])

# 计算5日、10日、15日和20日的滑动平均值
data['MA5'] = data['Close'].rolling(window=5).mean()
data['MA10'] = data['Close'].rolling(window=10).mean()
data['MA15'] = data['Close'].rolling(window=15).mean()
data['MA20'] = data['Close'].rolling(window=20).mean()

# 创建一个子图用于绘制MA5和MA10，并用填充色来表示它们的交叉区域
fig, (ax1, ax2, ax3) = plt.subplots(3, sharex=True, figsize=(12,12))
ax1.plot(data['MA5'], label='MA5', color='red')
ax1.plot(data['MA10'], label='MA10', color='green')
ax1.fill_between(data.index, data['MA5'], data['MA10'], where=data['MA5']>data['MA10'], color='red', alpha=0.3)
ax1.fill_between(data.index, data['MA5'], data['MA10'], where=data['MA5']<data['MA10'], color='green', alpha=0.3)
ax1.legend(loc='upper left')
ax1.set_title('MA5 and MA10 Crossover')

# 重复以上步骤，绘制 MA10和MA15、MA15和MA20的子图
ax2.plot(data['MA10'], label='MA10', color='blue')
ax2.plot(data['MA15'], label='MA15', color='orange')
ax2.fill_between(data.index, data['MA10'], data['MA15'], where=data['MA10']>data['MA15'], color='blue', alpha=0.3)
ax2.fill_between(data.index, data['MA10'], data['MA15'], where=data['MA10']<data['MA15'], color='orange', alpha=0.3)
ax2.legend(loc='upper left')
ax2.set_title('MA10 and MA15 Crossover')

ax3.plot(data['MA15'], label='MA15', color='purple')
ax3.plot(data['MA20'], label='MA20', color='brown')
ax3.fill_between(data.index, data['MA15'], data['MA20'], where=data['MA15']>data['MA20'], color='purple', alpha=0.3)
ax3.fill_between(data.index, data['MA15'], data['MA20'], where=data['MA15']<data['MA20'], color='brown', alpha=0.3)
ax3.legend(loc='upper left')
ax3.set_title('MA15 and MA20 Crossover')

plt.tight_layout()
plt.show()

# 定义EMA和MACD的计算周期
short_period = 12
long_period = 26
dea_period = 9

# 计算EMA12和EMA26（即对收盘价运用指数移动平均算法进行平滑）
data['EMA12'] = data['Close'].ewm(span=short_period).mean()
data['EMA26'] = data['Close'].ewm(span=long_period).mean()

# 计算DIF（Disparity Index，分离指数，即EMA12和EMA26的差值）
data['DIF'] = data['EMA12'] - data['EMA26']

# 计算DEA（DIF的移动平均值，用来确定交易信号）
data['DEA'] = data['DIF'].ewm(span=dea_period).mean()

# 计算MACD指标（红绿柱），基于DIF和DEA的差值
data['MACD'] = 2 * (data['DIF'] - data['DEA'])

# 基于MACD和DEA数值计算买入和卖出信号，这个策略中，MACD值大于0且DIF大于DEA表示买入，反之为卖出
data['Buy_Signal'] = np.where((data['MACD'] > 0) & (data['DIF'] > data['DEA']), data['Close'], np.nan)
data['Sell_Signal'] = np.where((data['MACD'] < 0) & (data['DIF'] < data['DEA']), data['Close'], np.nan)

# 将计算结果写入到一个新的Excel文件中，这可以用来复查或者再次使用这些数据
data.to_excel('output_data.xlsx', index=False)

# 基于MACD的正负，计算出价格的置信区间。该代码假设MACD大于0时市场是多头，小于0是空头，然后计算出了价格的论均值的1.96个标准差的信赖区间
data['Confidence_Interval'] = np.where(data['MACD'] > 0,
    data['Close'] + 1.96 * np.std(data['Close']),
    data['Close'] - 1.96 * np.std(data['Close'])
)

fig, (ax1, ax2) = plt.subplots(2, sharex=True, figsize=(12,8))

ax1.plot(data['Close'], label='Close Price', color='blue')
ax1.fill_between(data.index, data['Confidence_Interval'], color='gray', alpha=0.3)
ax1.plot(data.index, data['Buy_Signal'], marker='^', markersize=12, lw=0, color='g', label='buy')
ax1.plot(data.index, data['Sell_Signal'], marker='v', markersize=12, lw=0, color='r', label='sell')
ax1.legend(loc='upper left')
ax1.set_title('Close Price and Confidence Interval')

ax2.plot(data['DIF'], label='DIF', color='orange')
ax2.plot(data['DEA'], label='DEA', color='purple')
ax2.bar(data.index, data['MACD'], label='MACD', color='gray', alpha=0.3)
ax2.xaxis.set_major_locator(MultipleLocator(250)) 
ax2.legend(loc='upper left')
ax2.set_title('MACD Indicator')

plt.tight_layout()
plt.show()

# 将之前的买卖信号用1和-1标记，便于接下来计算统计信息
data['Buy_Signal_Flag'] = np.where((data['MACD'] > 0) & (data['DIF'] > data['DEA']), 1, 0)
data['Sell_Signal_Flag'] = np.where((data['MACD'] < 0) & (data['DIF'] < data['DEA']), -1, 0)

# 计算下一天的收盘价以及日收益率
data['Next_Day_Close'] = data['Close'].shift(-1)
data['Return'] = (data['Next_Day_Close'] - data['Close']) / data['Close']

# 计算交易次数（出现买卖信号即视为一次交易）
data['Trade_Flag'] = ((data['Buy_Signal_Flag'] == 1) | (data['Sell_Signal_Flag'] == -1))
trade_count = data['Trade_Flag'].sum()

# 计算盈利次数（买入后次日收益率大于0为盈利）
data['Profit_Flag'] = ((data['Buy_Signal_Flag'].shift(1) == 1) & (data['Return'] > 0))
profit_times = data['Profit_Flag'].sum()

# 计算胜率
win_rate = profit_times / trade_count

# 过滤掉收益率是inf、-inf和NaN的行
data = data.replace([np.inf, -np.inf], np.nan)
data = data.dropna(subset=['Return'], how='all')

# 计算累计收益率
data['Cumulative_Return'] = (1 + data['Return']).cumprod() - 1

# 定义一个计算最大回撤的函数
def calculate_max_drawdown(cumulative_returns):
    # 通过累计回报计算资金曲线，取1加上回报序列即可
    portfolio_values = 1 + cumulative_returns
    # 计算累计最高值
    running_max = portfolio_values.cummax()
    # 计算回撤
    drawdowns = 1 - portfolio_values / running_max
    # 计算最大回撤
    max_drawdown = drawdowns.max()

    return max_drawdown

# 删除含有NaN的行，为之后的计算提供清洁的数据
data = data.dropna(subset=['Cumulative_Return'])

# 使用上面的函数来计算最大回撤
max_drawdown = calculate_max_drawdown(data['Cumulative_Return'])

# 最后，输出模型表现的统计数据
print(f"Total number of trades: {trade_count}")
print(f"Total number of profitable trades: {profit_times}")
print(f"Win rate: {win_rate*100}%")
print(f"Cumulative return: {data['Cumulative_Return'].iloc[-1]*100}%")
print("The maximum drawdown : {:.2%}".format(max_drawdown))
